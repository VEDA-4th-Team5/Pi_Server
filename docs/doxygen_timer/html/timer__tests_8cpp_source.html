74template <typename Predicate>
75bool waitUntil(Predicate predicate, const std::chrono::milliseconds timeout) {
76 const auto deadline = std::chrono::steady_clock::now() + timeout;
77 while (std::chrono::steady_clock::now() < deadline) {
78 if (predicate()) {
79 return true;
80 }
81 std::this_thread::sleep_for(5ms);
82 }

181 EventDatabase database(path);
182 initialize(database);
183 const auto first_id = database.insertParked(
184 "123가4567", 1, parking_timer::utcNow(), "first.jpg");
185 const auto second_id = database.insertParked(
186 "234나5678", 2, parking_timer::utcNow(), "second.jpg");
187
188 std::mutex mutex;
189 std::condition_variable condition;
190 std::vector<std::int64_t> expired_order;
191 parking_timer::TimerManager timers(
192 database, [&](const parking_timer::ViolationEvent& event) {
193 {
194 std::lock_guard lock(mutex);
195 expired_order.push_back(event.log_id);
196 }
197 condition.notify_one();
198 });
199
200 // worker가 300ms를 기다리기 시작한 뒤 50ms 타이머를 넣어 notify/re-wait 경로를 탄다.
201 timers.schedule(first_id, 1, "123가4567", 300ms);
202 std::this_thread::sleep_for(20ms);
203 timers.schedule(second_id, 2, "234나5678", 50ms);
204
205 std::unique_lock lock(mutex);
206 require(condition.wait_for(lock, 500ms,
207 [&] { return !expired_order.empty(); }),
208 "priority queue worker did not wake for an earlier timer");
209 require(expired_order.front() == second_id,
210 "min-priority queue expired the later deadline first");
211 }
212 removeDatabaseFiles(path);
