364 try {
365 // partial unique index상 활성 행은 최대 하나지만, 최신 행을 명시적으로 선택한다.

367 db_,

369 "departed_at, image_path_1, image_path_2, is_canceled "
370 "FROM timer_log WHERE zone_id = ? AND departed_at IS NULL "
371 "ORDER BY id DESC LIMIT 1;");
372 find_statement.bindInt(1, zone_id);
373 const int find_result = sqlite3_step(find_statement.get());
374 if (find_result == SQLITE_DONE) {
375 executeSqlUnlocked("COMMIT;");
376 return std::nullopt;
377 }
378 if (find_result != SQLITE_ROW) {
379 throw std::runtime_error("SQLite active log lookup failed: " +
380 std::string(sqlite3_errmsg(db_)));
381 }
382 const auto log_id = sqlite3_column_int64(find_statement.get(), 0);
383
384 // is_canceled는 우선순위 큐 노드를 즉시 찾지 않고 나중에 버리기 위한 표시다.
385 Statement update_statement(
386 db_,
387 "UPDATE timer_log SET status = 'DEPARTS', departed_at = ?, "
388 "is_canceled = 1 WHERE id = ? AND departed_at IS NULL;");
389 update_statement.bindText(1, departed_at);
390 update_statement.bindInt64(2, log_id);
391 requireDone(db_, update_statement.get());
392 if (sqlite3_changes(db_) != 1) {
393 throw std::runtime_error("active parking log changed concurrently");

542void EventDatabase::executeSqlUnlocked(const std::string& sql) {

545 if (result == SQLITE_OK) {

550 sqlite3_free(raw_error);
