130void TimerManager::run() {
131 std::unique_lock lock(mutex_);
132 while (!stopping_) {
133 if (queue_.empty()) {
134 // polling하지 않고 첫 타이머 등록 또는 종료 신호가 올 때까지 잠든다.
135 cv_.wait(lock, [this] { return stopping_ || !queue_.empty(); });
136 continue;
137 }
138
139 const auto deadline = queue_.top().deadline;
140 const auto sequence = queue_.top().sequence;
141 // 기다리던 top을 캡처해 두고, 더 빠른 top이 새로 생긴 경우에만 대기 목표를 재계산한다.
142 // predicate 사용으로 spurious wake-up도 안전하게 다시 대기한다.
143 cv_.wait_until(lock, deadline, [this, deadline, sequence] {
144 return stopping_ || queue_.empty() ||
145 queue_.top().deadline < deadline ||
146 (queue_.top().deadline == deadline &&
147 queue_.top().sequence < sequence);
148 });
149 if (stopping_) {
150 break;
151 }
152
153 // 한 번 깨어났을 때 이미 만료된 동시 deadline 항목을 모두 연속으로 비운다.
154 while (!queue_.empty() && queue_.top().deadline <= Clock::now()) {
155 TimerItem expired = queue_.top();
156 queue_.pop();
157 // SQLite I/O와 사용자 callback이 느려도 새 입차 타이머 등록은 계속 가능해야 한다.
158 lock.unlock();
159 processExpired(std::move(expired));
160 lock.lock();
161 if (stopping_) {
162 break;
163 }
164 }
165 }

175void TimerManager::processExpired(TimerItem item) {
176 try {
177 std::unique_lock<std::mutex> transition_lock;
178 if (transition_mutex_ != nullptr) {
179 // 입차/출차의 DB 변경과 위반 UPDATE·이벤트 발행을 하나의 관찰 가능한 순서로 만든다.
180 transition_lock = std::unique_lock<std::mutex>{*transition_mutex_};
181 }
182
183 // DB 장애로 재시도하더라도 실제 최초 deadline 시각이 복구 시각으로 바뀌지 않게 고정한다.
184 if (item.violation_at.empty()) {
185 item.violation_at = utcNow();
186 }
187 const auto image_path = "snapshots/" + item.car_number + "_" +
188 std::to_string(item.log_id) + "_violation.jpg";
189
190 bool marked{};
191 try {
192 // 출차된 로그는 is_canceled=1이므로 UPDATE 대상이 0행이 된다.
193 // 이것이 명세의 log-id 기반 lazy deletion이다.
194 marked = database_.markViolation(item.log_id, item.violation_at, image_path);
195 } catch (const std::exception& error) {
196 // 재큐와 오류 발행 동안 다른 입출차까지 막지 않도록 외부 전이 lock을 먼저 놓는다.
197 if (transition_lock.owns_lock()) {
198 transition_lock.unlock();
199 }
200 retryAfterDatabaseError(std::move(item), error.what());
201 return;
202 } catch (...) {
203 if (transition_lock.owns_lock()) {
204 transition_lock.unlock();
205 }
206 retryAfterDatabaseError(std::move(item), "unknown SQLite failure");
207 return;
208 }
209 if (!marked) {
210 // 출차가 먼저 is_canceled=1로 만들었다면 이것이 의도한 lazy deletion 결과다.
211 return;
212 }
213
214 if (callback_) {
215 callback_(ViolationEvent{item.log_id, item.zone_id, item.car_number,
216 item.violation_at, image_path});
217 }
218 } catch (const std::exception& error) {
219 reportError(item, error.what());
220 } catch (...) {
221 reportError(item, "unknown timer worker failure");
222 }

233void TimerManager::retryAfterDatabaseError(TimerItem item, std::string message) noexcept {
234 reportError(item, "violation DB update failed; retry scheduled: " + message);
235 try {
236 // 일시적인 busy/I/O 장애가 위반 판정을 영구 유실시키지 않도록 1,2,4,8,16,32초
237 // 이후에는 32초 간격으로 재시도한다. 출차된 행은 복구 후 조건부 UPDATE 0행으로
238 // 자연스럽게 lazy-delete 된다.
239 const auto exponent = std::min(item.retry_count, std::uint32_t{5});
240 const auto retry_delay = std::chrono::seconds{std::int64_t{1} << exponent};
241 // retry_count를 5에서 포화시켜 장기 장애에서도 shift와 간격이 계속 커지지 않게 한다.
242 if (item.retry_count < std::uint32_t{5}) {
243 ++item.retry_count;
244 }
245 item.deadline = Clock::now() + retry_delay;
246
247 bool became_earliest{};
248 {
249 std::lock_guard lock(mutex_);
250 // 종료가 시작됐다면 소멸을 지연시키는 새 노드를 만들지 않는다.
251 if (stopping_) {
252 return;
253 }
254 item.sequence = next_sequence_++;
255 became_earliest = queue_.empty() || item.deadline < queue_.top().deadline;
256 queue_.push(std::move(item));
257 }
258 // 재시도 노드가 새 top이 된 경우 기존 wait_until을 다시 계산시킨다.

262 } catch (const std::exception& error) {
