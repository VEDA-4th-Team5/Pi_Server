48long long parsePositiveInteger(const std::string& value, const std::string& option) {
49 std::size_t consumed{};
50 long long parsed{};
51 try {
52 parsed = std::stoll(value, &consumed);
53 } catch (const std::exception&) {
54 throw std::invalid_argument(option + " requires a positive integer");
55 }
56 if (consumed != value.size() || parsed <= 0) {
57 throw std::invalid_argument(option + " requires a positive integer");
58 }
59 return parsed;
60}

70Options parseArguments(const int argc, char* argv[]) {
71 Options options;
72 for (int index = 1; index < argc; ++index) {
73 const std::string argument = argv[index];
74 // 값을 요구하는 옵션은 다음 argv를 소비하며, 없으면 현재 옵션 이름으로 진단한다.
75 auto requireValue = [&]() -> std::string {
76 if (index + 1 >= argc) {
77 throw std::invalid_argument(argument + " requires a value");
78 }
79 return argv[++index];
80 };
81
82 if (argument == "--config") {
83 options.config_file = requireValue();
84 } else if (argument == "--sql-dir") {
85 options.sql_directory = requireValue();
86 } else if (argument == "--db") {
87 options.database_path = requireValue();
88 } else if (argument == "--timeout-seconds") {
89 options.timeout = std::chrono::seconds{
90 parsePositiveInteger(requireValue(), argument)};
91 } else if (argument == "--demo") {
92 options.demo = true;
93 } else if (argument == "--reset-logs") {
94 options.reset_logs = true;
95 } else if (argument == "--help" || argument == "-h") {
96 options.help = true;
97 } else {
98 throw std::invalid_argument("unknown option: " + argument);
99 }
100 }

116std::filesystem::path findResource(
117 const std::optional<std::filesystem::path>& explicit_path,
118 const std::vector<std::filesystem::path>& candidates,
119 const std::string& description) {
120 if (explicit_path.has_value()) {
121 if (!std::filesystem::exists(*explicit_path)) {
122 throw std::runtime_error(description + " does not exist: " +
123 explicit_path->string());
124 }
125 return *explicit_path;
126 }
127 for (const auto& candidate : candidates) {
128 if (std::filesystem::exists(candidate)) {
129 return candidate;
130 }
131 }
132 throw std::runtime_error("cannot locate " + description);
133}

143std::filesystem::path executableDirectory(const char* argv0) {
144#if defined(__linux__)
145 std::error_code error;
146 const auto proc_path = std::filesystem::read_symlink("/proc/self/exe", error);

149 }
150#endif
